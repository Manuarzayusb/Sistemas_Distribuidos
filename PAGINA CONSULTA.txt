    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        try {
            // TODO add your handling code here:
            DefaultTableModel modelo = (DefaultTableModel)jTable1.getModel();
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost/systemahospital","root", "");
            Statement stmt = con.createStatement();
            
            ResultSet rs = stmt.executeQuery("SELECT * FROM pacientes WHERE documento="+jTextField1.getText());
            
            rs.first();
            
            do{
                String[] fila = {rs.getString(1),rs.getString(2),rs.getString(3),rs.getString(4),rs.getString(5)};
                modelo.addRow(fila);
            }while(rs.next());
            
        } catch (SQLException ex) {
