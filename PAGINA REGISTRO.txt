    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        int year = jDateChooser1.getCalendar().get(Calendar.YEAR);
        int month = jDateChooser1.getCalendar().get(Calendar.MONTH);
        int day = jDateChooser1.getCalendar().get(Calendar.DAY_OF_MONTH);
        String fechanacimiento = ""+year+"-"+month+"-"+day;
        String sexo="";
        
        if(jRadioButton1.isSelected()){
            sexo="M";
        }if (jRadioButton2.isSelected()){
            sexo="F";
        }
        
        try {
            con = DriverManager.getConnection("jdbc:mysql://localhost/systemahospital","root", "");
            Statement stmt = con.createStatement();
            stmt.executeUpdate("INSERT INTO pacientes VALUES ('"+jTextField1.getText()+"','"+jTextField2.getText()+"','"+jTextField3.getText()+"','"+fechanacimiento+"','"+sexo+"')");
        } catch (SQLException ex) {
            Logger.getLogger(pantallaregistro.class.getName()).log(Level.SEVERE, null, ex);
            JOptionPane.showMessageDialog(null, "No se puede conectar a la base de datos");
        }
        JOptionPane.showMessageDialog(null, "El paciente ha sido registrado demanera existosa.");
    }
